AnimDialog: 动画对话框组件

  • 淡入淡出对话框: fade
seajs.use(['anim-dialog'], function(AnimDialog) {
    var closeDialogTpl = '<div class="overlay"><button id="close">点击关闭</button><p>肯定是房间里萨的看法金克拉束带结发</p></div>';

    var d2 = new AnimDialog({
        trigger: '#trigger2',
        template: closeDialogTpl,
        width: 300,
        height: 200,
        closeElement: '#close',
        align: {
            baseXY: [400, 200]
        },
        effect: {
            type: 'fade'
        }
    });
});

  • 混合动画对话框
seajs.use(['anim-dialog'], function(AnimDialog) {
    var closeDialogTpl = '<div class="overlay"><button id="close">点击关闭</button><p>肯定是房间里萨的看法金克拉束带结发</p></div>';
    var d9 = new AnimDialog({
        trigger: '#trigger9',
        template: closeDialogTpl,
        width: 300,
        height: 200,
        closeElement: '#close',
        align: {
            baseXY: [400, 200]
        },
        showEffect: {
            type: 'move',
            from: 'down'
        },
        hideEffect: {
            type: 'fade'
        }
    });
    var d10 = new AnimDialog({
        trigger: '#trigger10',
        template: closeDialogTpl,
        width: 300,
        height: 200,
        closeElement: '#close',
        align: {
            baseXY: [400, 200]
        },
        showEffect: {
            type: 'none'
        },
        hideEffect: {
            type: 'move',
            from: 'left'
        }
    });
});
