seajs.use(['base-dialog'], function(BaseDialog) {
    var d1 = new BaseDialog({
        trigger: '#trigger1',
        element: '#confirmBox',
        width: 300,
        title: '我是标题',
        content: '我是内容',
        onConfirm: function() {
            alert('点击了确定按钮');
        },
        onClose: function() {
            alert('点击了关闭按钮');
        },
        align: {
            baseXY: [200, 200]
        },
        hasMask: true
    });
    d1.set('content', '改变的内容');
    d1.set('width', 500);
});
